FTP/-E ERIS Directory/C LispCore>Next Ret/<>A Lisp.run DoradoLispMc.eb;

// start up DLInit. MAKESYSNAME gets set here. The (SPECVARS . T) is because some file in the loadup (unknown) sets it wrong

(SETQQ MAKESYSNAME LISPCORE)
(LOAD (QUOTE {ERIS}<LISPCORE>SOURCES>LOADUP.LISP))
(HARDRESET] 
SHH(ENDLOADUP] 
SHH(SPECVARS . T]
SHH(* MAKESYS '{ERIS}<LISPCORE>NEXT>LISP.SYSOUT]  
SHH(LOAD '{ERIS}<LISPCORE>SOURCES>LOADFULL.LISP)
SHH(MAKESYS '{ERIS}<LISPCORE>NEXT>FULL.SYSOUT]
SHH(LOGOUT T] "
