Set[PIXELBank, 02];
Set[PSCLRBank, 02];

{@PIXELOP:	opcode[305'b],
	Bank ← PIXELBank, 		c1; {For Pixelblt}
	uTOS ← TOS,			c2;
	uTOSH ← TOSH, CROSS[PIXEL],	c3;}
	
		
@PSCLROP:	opcode[70'b],
	Bank ← PSCLRBank, 		c1; {For Pseudo8bitColor}
	uTOS ← TOS,			c2;
	uTOSH ← TOSH, CROSS[PSCLR],	c3;
	
	{END}
