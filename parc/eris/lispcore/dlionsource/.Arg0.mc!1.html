This program check if the displacement between PV and IVAR is
correct
}

@VARCH:	TT ← uIVar,				c1, opcode[141'b];
	Ybus ← TOSH xor smallpl, ZeroBr,	c2;
	Rx ← TOS - 1, BRANCH[VAER0, $],		c3;
	
	Rx ← Rx LShift1, SE ← 0,		c1;	
	Rx ← Rx + TT, L3 ← 0,			c2;
	TT ← Rx + 12'd,	L2 ← 0,			c3;
	
	TT ← PV - TT, CarryBr,			c1;
		
	BRANCH[IVX, VAER1],		c2;
	
VAER0:	GOTO[ufnX2],			c1;

VAER1:	GOTO[ufnX1],			c3;

IVX:	rhRx ← nRhS, GOTO[Var2],	c3;


	{END}
