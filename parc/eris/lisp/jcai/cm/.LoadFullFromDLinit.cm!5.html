// bring over files necessary to run INIT.DLINIT. This command file works on Dorados
FTP/-E ERIS Directory/C Lisp>JCAI>basics Ret/<>A Lisp.run DoradoLispMc.eb;

// start up DLInit. MAKESYSNAME gets set here. The (SPECVARS . T) is because some file in the loadup (unknown) sets it wrong

Lisp [ERIS]<Lisp>JCAI>basics>Init.dlinit ;"
LOADUP(HUGE](SETQQ MAKESYSNAME JCAI)(HARDRESET]
SHH(ENDLOADUP]
SHH(SPECVARS . T]
SHH(MAKESYS '{ERIS}<LISP>JCAI>basics>LISP.SYSOUT]
SHH(RESETVARS (( IDLE.OPTIONS '(UNLOCKED NEVER)))
(SETQ DISPLAYFONTDIRECTORIES (SETQQ INTERPRESSFONTDIRECTORIES ({ERIS}<LISP>FONTS>)))
(SETQQ PRESSFONTWIDTHSFILES ({ERIS}<LISP>FONTS>FONTS.WIDTHS))
(LOADUP (QUOTE (HELPSYS CHAT PUPCHAT NSCHAT TEDIT HRULE TEDITCHAT READNUMBER EDITBITMAP MAILCLIENT GRAPEVINE NSMAIL LAFITE FILEBROWSER TELERAID GRAPHER SPY AREDIT WHEREIS COPYFILES))))
SHH(ENDLOADUP)
SHH(MAKESYS '{ERIS}<LISP>JCAI>basics>FULL.SYSOUT]
(LOGOUT T]"
