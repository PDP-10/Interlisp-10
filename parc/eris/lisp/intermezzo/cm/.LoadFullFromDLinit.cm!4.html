// bring over files necessary to run INIT.DLINIT. This command file works on both Dolphins and Dorados
FTP/-E ERIS Directory/C Lisp>Intermezzo>Basics Ret/<>A Lisp.run DoradoLispMc.eb DolphinLispMc.eb;

// start up DLInit. MAKESYSNAME gets set here. The (SPECVARS . T) is because some file in the loadup (unknown) sets it wrong

Lisp [ERIS]<LISP>INTERMEZZO>BASICS>Init.dlinit ;"
LOADUP(HUGE]
(SETQQ MAKESYSNAME INTERMEZZO)
(HARDRESET] 
SHH(ENDLOADUP] 
SHH(SPECVARS . T]
SHH(SETQQ CAR/CDRERR NIL]
SHH(SETQQ CROSSCOMPILING ASK]
SHH(MAKESYS '{ERIS}<LISP>INTERMEZZO>BASICS>LISP.SYSOUT]  
SHH(SETQQ DISPLAYFONTDIRECTORIES ({ERIS}<LISP>FONTS> {INDIGO}<ALTOFONTS> {INDIGO}<ALTOFONTS>ORIGINAL>))
SHH(SETQQ DISPLAYFONTEXTENSIONS (DISPLAYFONT STRIKE AC))
SHH(SETQQ PRESSFONTWIDTHSFILES ({ERIS}<LISP>FONTS>FONTS.WIDTHS {INDIGO}<FONTS>FONTS.WIDTHS))
SHH(SETQQ INTERPRESSFONTDIRECTORIES ({ERIS}<LISP>FONTS> {INDIGO}<ALTOFONTS> {INDIGO}<ALTOFONTS>ORIGINAL>))
LOADUP((DPUPFTP TEDIT READNUMBER EDITBITMAP MAILCLIENT GRAPEVINE LAFITE MTP FILEBROWSER TELERAID GRAPHER SPY SINGLEFILEINDEX AREDIT WHEREIS]
ENDLOADUP]
SHH(MAKESYS '{ERIS}<LISP>INTERMEZZO>BASICS>FULL.SYSOUT]
SHH(LOGOUT] "
